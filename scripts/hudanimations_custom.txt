event HudHealthBonusPulse
{
	Animate	HealthBG					BgColor		"Overheal"			Linear 0.0 0.2
	Animate	HealthBG					BgColor		"OverhealPulse"		Linear 0.2 0.6

//	Animate	PlayerHealthValue			FgColor		"Overheal"			Linear 0.0 0.2
//	Animate	PlayerHealthValue			FgColor		"OverhealPulse"		Linear 0.2 0.6

	Animate	PlayerGuiHealthValue		FgColor		"PositiveValue"		Linear 0.0 0.0

	RunEvent HudHealthBonusPulseLoop	0.7
}

event HudHealthBonusPulseStop
{
	StopEvent HudHealthBonusPulse 		0.0
	StopEvent HudHealthBonusPulseLoop	0.0
	
	Animate	HealthBG					BgColor		"Blank"				Linear 0.0 0.0

	Animate PlayerHealthValue			FgColor		"HudWhite"			Linear 0.0 0.0
	Animate PlayerGuiHealthValue		FgColor		"HudWhite"			Linear 0.0 0.0
}
